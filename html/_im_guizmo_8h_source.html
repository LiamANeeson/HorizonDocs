170 enum OPERATION
171 {
172 NONE = 0u,
173 TRANSLATE_X = (1u << 0),
174 TRANSLATE_Y = (1u << 1),
175 TRANSLATE_Z = (1u << 2),
176 ROTATE_X = (1u << 3),
177 ROTATE_Y = (1u << 4),
178 ROTATE_Z = (1u << 5),
179 ROTATE_SCREEN = (1u << 6),
180 SCALE_X = (1u << 7),
181 SCALE_Y = (1u << 8),
182 SCALE_Z = (1u << 9),
183 BOUNDS = (1u << 10),
184 SCALE_XU = (1u << 11),
185 SCALE_YU = (1u << 12),
186 SCALE_ZU = (1u << 13),
187
188 TRANSLATE = TRANSLATE_X | TRANSLATE_Y | TRANSLATE_Z,
189 ROTATE = ROTATE_X | ROTATE_Y | ROTATE_Z | ROTATE_SCREEN,
190 SCALE = SCALE_X | SCALE_Y | SCALE_Z,
191 SCALEU = SCALE_XU | SCALE_YU | SCALE_ZU, // universal
192 UNIVERSAL = TRANSLATE | ROTATE | SCALEU
193 };

195 inline OPERATION operator|(OPERATION lhs, OPERATION rhs)

197 return static_cast<OPERATION>(static_cast<int>(lhs) | static_cast<int>(rhs));
