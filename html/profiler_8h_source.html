173/*
174 * The macros below will generate the majority of the callback API. Refer to
175 * mono/metadata/profiler-events.h for a list of callbacks. They are expanded
176 * like so:
177 *
178 * typedef void (*MonoProfilerRuntimeInitializedCallback (MonoProfiler *prof);
179 * MONO_API void mono_profiler_set_runtime_initialized_callback (MonoProfiler *prof, MonoProfilerRuntimeInitializedCallback cb);
180 *
181 * typedef void (*MonoProfilerRuntimeShutdownCallback (MonoProfiler *prof);
182 * MONO_API void mono_profiler_set_runtime_shutdown_callback (MonoProfiler *prof, MonoProfilerRuntimeShutdownCallback cb);
183 *
184 * typedef void (*MonoProfilerContextLoadedCallback (MonoProfiler *prof);
185 * MONO_API void mono_profiler_set_context_loaded_callback (MonoProfiler *prof, MonoProfilerContextLoadedCallback cb);
186 *
187 * typedef void (*MonoProfilerContextUnloadedCallback (MonoProfiler *prof);
188 * MONO_API void mono_profiler_set_context_unloaded_callback (MonoProfiler *prof, MonoProfilerContextUnloadedCallback cb);
189 *
190 * Etc.
191 *
192 * To remove a callback, pass NULL instead of a valid function pointer.
193 * Callbacks can be changed at any point, but note that doing so is inherently
194 * racy with respect to threads that aren't suspended, i.e. you may still see a
195 * call from another thread right after you change a callback.
196 *
197 * These functions are async safe.
198 */
199
