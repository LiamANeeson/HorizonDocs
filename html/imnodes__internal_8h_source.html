375template<typename T>
376static inline void ObjectPoolUpdate(ImObjectPool<T>& objects)
377{
378 for (int i = 0; i < objects.InUse.size(); ++i)
379 {
380 const int id = objects.Pool[i].Id;
381
382 if (!objects.InUse[i] && objects.IdMap.GetInt(id, -1) == i)
383 {
384 objects.IdMap.SetInt(id, -1);
385 objects.FreeList.push_back(i);
386 (objects.Pool.Data + i)->~T();
387 }
388 }
389}
390
391template<>
392inline void ObjectPoolUpdate(ImObjectPool<ImNodeData>& nodes)
393{
394 for (int i = 0; i < nodes.InUse.size(); ++i)
395 {
396 if (nodes.InUse[i])
397 {
398 nodes.Pool[i].PinIndices.clear();
399 }
400 else
401 {
402 const int id = nodes.Pool[i].Id;
403
404 if (nodes.IdMap.GetInt(id, -1) == i)
405 {
406 // Remove node idx form depth stack the first time we detect that this idx slot is
407 // unused
408 ImVector<int>& depth_stack = EditorContextGet().NodeDepthOrder;
409 const int* const elem = depth_stack.find(i);
410 IM_ASSERT(elem != depth_stack.end());
411 depth_stack.erase(elem);
412
413 nodes.IdMap.SetInt(id, -1);
414 nodes.FreeList.push_back(i);
415 (nodes.Pool.Data + i)->~ImNodeData();
416 }
417 }
418 }
419}

430template<typename T>
431static inline int ObjectPoolFindOrCreateIndex(ImObjectPool<T>& objects, const int id)
432{
433 int index = objects.IdMap.GetInt(static_cast<ImGuiID>(id), -1);
434
435 // Construct new object
436 if (index == -1)
437 {
438 if (objects.FreeList.empty())
439 {
440 index = objects.Pool.size();
441 IM_ASSERT(objects.Pool.size() == objects.InUse.size());
442 const int new_size = objects.Pool.size() + 1;
443 objects.Pool.resize(new_size);
444 objects.InUse.resize(new_size);
445 }
446 else
447 {
448 index = objects.FreeList.back();
449 objects.FreeList.pop_back();
450 }
451 IM_PLACEMENT_NEW(objects.Pool.Data + index) T(id);
452 objects.IdMap.SetInt(static_cast<ImGuiID>(id), index);
453 }
454
455 // Flag it as used
456 objects.InUse[index] = true;
457
458 return index;
459}

461template<>
462inline int ObjectPoolFindOrCreateIndex(ImObjectPool<ImNodeData>& nodes, const int node_id)
463{
464 int node_idx = nodes.IdMap.GetInt(static_cast<ImGuiID>(node_id), -1);
465
466 // Construct new node
467 if (node_idx == -1)

471 node_idx = nodes.Pool.size();

473 const int new_size = nodes.Pool.size() + 1;
474 nodes.Pool.resize(new_size);
475 nodes.InUse.resize(new_size);
